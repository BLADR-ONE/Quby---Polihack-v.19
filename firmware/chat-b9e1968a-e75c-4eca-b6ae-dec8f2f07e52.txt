#include <Arduino.h>
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

// =====================================================
// 1. PIN CONFIGURATION
// =====================================================

// OLED pins
#define SDA_PIN 8
#define SCL_PIN 9

// Buzzer pin
#define BUZZER_PIN 0

// RGB pins
#define RED_PIN   10
#define GREEN_PIN 6
#define BLUE_PIN  3

// MQ135 analog pin
#define MQ135_PIN 4

// OLED config
#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_ADDR 0x3C

// =====================================================
// 2. GLOBAL OBJECTS
// =====================================================

Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, -1);

// =====================================================
// 3. ROBOT FACE SETTINGS
// =====================================================

int leftEyeX = 42;
int rightEyeX = 86;
int eyeY = 30;

// =====================================================
// 4. MQ135 / AIR QUALITY SETTINGS
// =====================================================

// MQ135 nu este senzor CO2 real in ppm.
// Il folosim ca air quality / CO2-like indicator.
const unsigned long MQ_WARMUP_MS = 20000;
const unsigned long SENSOR_READ_MS = 1000;

float mq135Baseline = 1.0;
int mq135Raw = 0;
float mq135Index = 1.0;

unsigned long bootMs = 0;
unsigned long lastSensorReadMs = 0;

// Praguri relative fata de baseline
const float AIR_HAPPY_MAX = 1.20;
const float AIR_NEUTRAL_MAX = 1.65;

// =====================================================
// 5. MOOD STATE
// =====================================================

enum Mood {
  HAPPY,
  NEUTRAL,
  SAD
};

Mood currentMood = NEUTRAL;
Mood lastMood = NEUTRAL;

String reasonText = "Starting";

// =====================================================
// 6. RGB MODULE
// =====================================================

void setColor(int red, int green, int blue) {
  analogWrite(RED_PIN, red);
  analogWrite(GREEN_PIN, green);
  analogWrite(BLUE_PIN, blue);
}

void rgbHappy() {
  setColor(0, 255, 0);     // verde
}

void rgbNeutral() {
  setColor(0, 0, 255);     // albastru
}

void rgbSad() {
  setColor(255, 0, 0);     // rosu
}

void rgbOff() {
  setColor(0, 0, 0);
}

// =====================================================
// 7. BUZZER MODULE
// =====================================================

void playTone(int frequency, int duration) {
  tone(BUZZER_PIN, frequency);
  delay(duration);
  noTone(BUZZER_PIN);
}

void playHappySound() {
  playTone(660, 100);
  delay(40);
  playTone(880, 100);
  delay(40);
  playTone(1046, 150);
  delay(60);
  playTone(1320, 180);
}

void playNeutralSound() {
  playTone(700, 120);
  delay(80);
  playTone(700, 120);
}

void playSadSound() {
  playTone(600, 180);
  delay(70);
  playTone(440, 220);
  delay(70);
  playTone(330, 300);
}

void playBlinkSound() {
  playTone(1000, 60);
}

void playMoveSound() {
  playTone(850, 60);
}

void playMoodSound(Mood mood) {
  if (mood == HAPPY) {
    playHappySound();
  } else if (mood == NEUTRAL) {
    playNeutralSound();
  } else {
    playSadSound();
  }
}

// =====================================================
// 8. OLED FACE DRAWING
// =====================================================

void drawTriangleEye(int cx, int cy, bool inverted) {
  if (!inverted) {
    // happy: varful in sus
    display.fillTriangle(
      cx, cy - 14,
      cx - 15, cy + 12,
      cx + 15, cy + 12,
      SSD1306_WHITE
    );
  } else {
    // sad: varful in jos
    display.fillTriangle(
      cx - 15, cy - 12,
      cx + 15, cy - 12,
      cx, cy + 14,
      SSD1306_WHITE
    );
  }
}

void drawClosedEye(int cx, int cy) {
  display.drawLine(cx - 16, cy, cx + 16, cy, SSD1306_WHITE);
  display.drawLine(cx - 16, cy + 1, cx + 16, cy + 1, SSD1306_WHITE);
}

void drawNeutralEye(int cx, int cy) {
  display.fillRoundRect(
    cx - 20,
    cy - 7,
    40,
    14,
    4,
    SSD1306_WHITE
  );
}

void drawHappyFace(int offsetX = 0, int offsetY = 0) {
  display.clearDisplay();

  drawTriangleEye(leftEyeX + offsetX, eyeY + offsetY, false);
  drawTriangleEye(rightEyeX + offsetX, eyeY + offsetY, false);

  display.display();
  rgbHappy();
}

void drawSadFace(int offsetX = 0, int offsetY = 0) {
  display.clearDisplay();

  drawTriangleEye(leftEyeX + offsetX, eyeY + offsetY, true);

void drawBlink() {
  display.clearDisplay();

  drawClosedEye(leftEyeX, eyeY);
  drawClosedEye(rightEyeX, eyeY);

  display.display();
}

  display.clearDisplay();

  drawTriangleEye(leftEyeX, eyeY - 5, false);

  display.setTextSize(1);

// =====================================================
// 9. OLED MESSAGE SCREEN
// =====================================================

void drawAirMessage(String title, String message) {
  display.clearDisplay();

  display.setTextColor(SSD1306_WHITE);
  display.setTextSize(1);

  display.setCursor(0, 0);
  display.println(title);

  display.setCursor(0, 15);

  display.setCursor(0, 35);

// =====================================================
// 10. MQ135 SENSOR MODULE
// =====================================================

int readMqAverage() {
  long total = 0;

  for (int i = 0; i < 10; i++) {
    total += analogRead(MQ135_PIN);
    delay(3);
  }

  return total / 10;
}

void readMQ135() {
  mq135Raw = readMqAverage();

  // In perioada de warmup invatam baseline-ul aerului curent
  if (millis() - bootMs < MQ_WARMUP_MS) {
    mq135Baseline = mq135Baseline * 0.90 + mq135Raw * 0.10;
  }

  if (mq135Baseline < 1.0) {
    mq135Baseline = 1.0;
  }

  mq135Index = mq135Raw / mq135Baseline;
}

// =====================================================
// 11. MOOD LOGIC FROM AIR QUALITY
// =====================================================

void evaluateAirMood() {
  Mood previousMood = currentMood;

  if (millis() - bootMs < MQ_WARMUP_MS) {
    currentMood = NEUTRAL;
    reasonText = "Sensor warmup";
  } else if (mq135Index <= AIR_HAPPY_MAX) {
    currentMood = HAPPY;
    reasonText = "Air very good";
  } else if (mq135Index <= AIR_NEUTRAL_MAX) {
    currentMood = NEUTRAL;
    reasonText = "Air acceptable";
  } else {
    currentMood = SAD;
    reasonText = "Air quality bad";
  }

  if (currentMood != previousMood) {
    playMoodSound(currentMood);
  }
}

// =====================================================
// 12. OUTPUT SYNC
// =====================================================

void updateMoodOutput() {
  if (currentMood == HAPPY) {
    drawHappyFace();
    rgbHappy();
  } else if (currentMood == NEUTRAL) {
    drawNeutralFace();
    rgbNeutral();
  } else {
    drawSadFace();
    rgbSad();
  }
}

String moodTitle() {
  if (currentMood == HAPPY) return "Quby is happy";
  if (currentMood == NEUTRAL) return "Quby is neutral";
  return "Quby is sad";
}

// =====================================================
// 13. SERIAL DEBUG
// =====================================================

void printMqDebug() {
  Serial.print("MQ135 raw: ");
  Serial.print(mq135Raw);

  Serial.print(" | baseline: ");
  Serial.print(mq135Baseline);

  Serial.print(" | index: ");
  Serial.print(mq135Index, 2);

  Serial.print(" | mood: ");
  if (currentMood == HAPPY) Serial.print("HAPPY");
  else if (currentMood == NEUTRAL) Serial.print("NEUTRAL");
  else Serial.print("SAD");

  Serial.print(" | reason: ");
  Serial.println(reasonText);
}

// =====================================================
// 14. SETUP
// =====================================================

void setup() {
  Serial.begin(115200);
  delay(500);

  bootMs = millis();

  // Buzzer
  pinMode(BUZZER_PIN, OUTPUT);
  noTone(BUZZER_PIN);

  // RGB
  pinMode(RED_PIN, OUTPUT);
  pinMode(GREEN_PIN, OUTPUT);
  pinMode(BLUE_PIN, OUTPUT);
  rgbOff();

  // MQ135
  pinMode(MQ135_PIN, INPUT);

#if defined(ARDUINO_ARCH_ESP32)
  analogReadResolution(12);
  analogSetPinAttenuation(MQ135_PIN, ADC_11db);
#endif

  // I2C + OLED
  Wire.begin(SDA_PIN, SCL_PIN);

  if (!display.begin(SSD1306_SWITCHCAPVCC, OLED_ADDR)) {
    Serial.println("OLED nu a pornit. Verifica SDA/SCL/VCC/GND sau adresa 0x3C.");
    while (true);
  }

  Serial.println("OLED + BUZZER + RGB + MQ135 pornite OK!");

  display.clearDisplay();
  display.display();

  // Initial baseline
  mq135Raw = readMqAverage();
  mq135Baseline = max(1.0f, (float)mq135Raw);

  drawText("Quby awake");
  playHappySound();
  delay(1200);
}

// =====================================================
// 15. LOOP
// =====================================================

void loop() {
  if (millis() - lastSensorReadMs >= SENSOR_READ_MS) {
    lastSensorReadMs = millis();

    // 1. Citim senzorul
    readMQ135();

    // 2. Decidem starea
    evaluateAirMood();

    // 3. Debug in Serial Monitor
    printMqDebug();

    // 4. Sincronizam OLED + RGB
    updateMoodOutput();
    delay(1800);

    // 5. Afisam motivul si valorile
    drawAirMessage(moodTitle(), reasonText);
    delay(1800);
  }
}
